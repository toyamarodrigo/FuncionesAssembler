COMPLE PROC	
PUSH AX
PUSH DX

mov SI, 8
mov DX, CX
mov AH, 9
int 21H
	XOR BL, 0FFH
	ADD BL, 1
CICLO2:
	SHL BL, 1
	JC IMPUNO
IMPCERO: 
	mov AH, 6
	mov DL, '0'
	int 21h
	JMP SIGO1
IMPUNO:
	mov AH, 6
	mov DL, '1'
	int 21h
SIGO1:
	DEC SI
	cmp SI, 0
	JNE CICLO2

POP AX

endp
