; xRegistro - Funcion Impresion HEX 
;Recibe en CL el HEX a imprimir

IMPRIMIRHEX PROC
PUSH AX
PUSH BX
PUSH DX

CIFRA1:	
	MOV BL, CL
	SHR BL, 4
	CMP BL, 9
	JLE IMPNUM1
IMPLETRA1:
	ADD BL, 55
	MOV DL, BL
	MOV AH, 6
	INT 21H
	JMP CIFRA2
IMPNUM1:
	ADD BL, 48
	MOV DL, BL
	MOV AH, 6
	INT 21H	
CIFRA2:
	AND CL, 0FH
	CMP CL, 9
	JLE IMPNUM2
IMPLETRA2:
	ADD CL, 55
	MOV DL, CL
	MOV AH, 6
	INT 21H
	JMP FIN
IMPNUM2:
	ADD CL, 48
	MOV DL, CL
	MOV AH, 6
	INT 21H
FIN:	
    POP DX
    POP BX
    POP AX
RET
ENDP
